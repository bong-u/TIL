  • 파스칼의 사각형? dp 구현

    N, K = map(int, input().split())

if K == 0:
	print(1)
	exit()

dp = [[0]*N for _ in range(K)]

for i in range(N):
	dp[0][i] = i+1

for i in range(1, K):
	for j in range(i, N):
		dp[i][j] = (dp[i-1][j-1] + dp[i][j-1]) % 10007

print (dp[K-1][N-1])

1309 : 동물원 (S1)

CREATED 2023-01-12

MODIFIED 2023-02-05

N = int(input())
dp = [1, 3] + [0]*(N-1)

for i in range(2, N+1):
    dp[i] = (dp[i-1]*2 + dp[i-2]) % 9901

print(dp[N])

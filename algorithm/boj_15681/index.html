 2sys.setrecursionlimit(10**6)
 3input = sys.stdin.readline
 4N, R, Q = map(int, input().split())
 5G = [[] for _ in range(N)]
 6cnt = [1]*N
 7visited = [False]*N
 8
 9for _ in range(N-1):
10    a, b = map(int, input().split())
11    G[a-1].append(b-1)
12    G[b-1].append(a-1)
13
14def dfs(node):
15    visited[node] = True
16
17    for i in G[node]:
18        if not visited[i]:
19            cnt[node] += dfs(i)
20    return cnt[node]

22dfs(R-1)

24for _ in range(Q):

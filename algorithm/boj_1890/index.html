수정일 : 2023-12-27

 1N = int(input())
 2
 3L = [list(map(int, input().split())) for _ in range(N)]
 4dp = [[0]*N for _ in range(N)]
 5dp[0][0] = 1
 6for i in range(N):
 7    for j in range(N):
 8        if L[i][j] == 0:
 9            continue
10        if dp[i][j] != 0:
11            right = j+L[i][j]
12            bottom = i+L[i][j]
13
14            if right < N:
15                dp[i][right] += dp[i][j]
16            if bottom < N:
17                dp[bottom][j] += dp[i][j]
18print (dp[N-1][N-1])

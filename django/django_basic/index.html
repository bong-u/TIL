12python manage.py test
13# 테스트용 서버를 특정 설정으로 실행
14python manage.py testserver

Paste Model 정의

 1from django.db import models
 2
 3class Paste(models.Model):
 4    title = models.CharField(max_length=100)
 5    content = models.TextField()
 6    # auto_now_add : 객체가 처음 생성될 때만 현재 날짜와 시간을 자동으로 설정
 7    created_at = models.DateTimeField(auto_now_add=True)
 8    # auto_now : 객체가 저장될 때마다 현재 날짜와 시간을 자동으로 설정
 9    updated_at = models.DateTimeField(auto_now=True)
10    class Meta:
11        ordering = ['-created_at']
12    def __str__(self):
13        return self.title

PasteView - get 구현

 1from rest_framework.views import APIView
 2from rest_framework.response import Response
 3from rest_framework import status, permissions
 4from .models import Paste
 5from .serializers import PasteSerializer
 6
 7class PasteView(APIView):
 8    # 모든 사용자에게 접근을 허용
 9    permission_classes = [permissions.AllowAny]
10
11    def get(self, request):
12        pastes = Paste.objects.all()
13        serializer = PasteSerializer(pastes, many=True)
14        return Response(serializer.data, status=status.HTTP_200_OK)
